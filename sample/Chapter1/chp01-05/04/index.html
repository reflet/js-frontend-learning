  function AppView(el) {
    this.initialize(el);
    this.handleEvents();
  }

  AppView.prototype.initialize = function(el) {
    this.$el = $(el);
    var obj = this.$el.data();
    if (this.$el.prop("required")) {
      obj["required"] = true;
    }

    this.model = new AppModel(obj);
  };

  AppView.prototype.handleEvents = function() {
    var self = this;
    this.$el.on("keyup", function(e) {
      self.onKeyup(e);
    });
  };
